MAIN_PROG CODE                      ; let linker place main program

START
	BSF STATUS, RP0
	BSF TRISA, 1
	BANKSEL ANSEL
	CLRF ANSEL
	BANKSEL TRISB
	BCF TRISB, 1
	BCF STATUS, RP0
    
; ***INICIO DEL PROGRAMA*** ;
INICIO    
	BTFSS PORTA, 1     ;PREGUNTA SI HAY UN PULSO ALTO (1) EN LA ENTRADA RA1
	GOTO INICIO        ;SI NO DETECTA UN PULSO ALTO (1) Y DETECTA PULSO BAJO (0) REGRESA A LA ETIQUETA INICIO
	BSF PORTB, 1       ;SI DETECTA UN PULSO ALTO, MANDA UN PULSO A LA SALIDA
	GOTO APAGAR        ;
APAGAR	BTFSC PORTA, 1     ;PREGUNTA POR UN PULSO BAJO (0) EN LA ENTRADA RA1
	GOTO APAGAR        ;SI NO DETECTA UN PULSO BAJO (0) Y DETECTA UN PULSO ALTO (1) REGRESA A LA ETIQUETA APAGAR
	BCF PORTB, 1       ;SI DETECTO UN PULSO BAJO, APAGA EL LED
	GOTO INICIO
	END
